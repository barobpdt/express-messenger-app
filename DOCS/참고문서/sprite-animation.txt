import { Rembg } from "rembg-node";
import fs from "fs";

async function removeBackground(inputPath, outputPath) {
    const input = fs.readFileSync(inputPath);
    // 최초 실행 시 모델(.onnx)을 다운로드하여 약간의 시간이 걸립니다.
    const rembg = new Rembg(); 
    const output = await rembg.remove(input); // AI 추론 진행
    fs.writeFileSync(outputPath, output.base64Image, 'base64');
}
