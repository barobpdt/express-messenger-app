import archiver from 'archiver';
import fs from 'fs';
import path from 'path';

/**
 * 폴더를 ZIP으로 압축
 * @param {string} sourceDir - 압축할 폴더 경로
 * @param {string} outputPath - 출력 ZIP 파일 경로
 */
function zipFolder(sourceDir, outputPath) {
  return new Promise((resolve, reject) => {
    const output = fs.createWriteStream(outputPath);
    const archive = archiver('zip', { zlib: { level: 9 } }); // 최대 압축

    output.on('close', () => {
      console.log(`✅ ZIP 완료: ${archive.pointer()} bytes`);
      resolve(outputPath);
    });
    archive.on('error', reject);

    archive.pipe(output);
    archive.directory(sourceDir, false); // false = 루트 폴더명 제외
    archive.finalize();
  });
}

// 사용 예
await zipFolder('./public', './backup/public.zip');

import archiver from 'archiver';

// GET /api/download?dir=uploads
app.get('/api/download', (req, res) => {
  const dir = req.query.dir || 'uploads';
  const safePath = path.join(__dirname, dir);

  // 경로 탈출 방지
  if (!safePath.startsWith(__dirname)) {
    return res.status(400).json({ error: 'Invalid path' });
  }

  res.setHeader('Content-Type', 'application/zip');
  res.setHeader('Content-Disposition', `attachment; filename="${dir}.zip"`);

  const archive = archiver('zip', { zlib: { level: 6 } });
  archive.on('error', err => res.status(500).send(err.message));
  archive.pipe(res);
  archive.directory(safePath, dir);
  archive.finalize();
});



import fs from 'fs';
import path from 'path';

// 재귀적으로 파일 목록 수집
function walkDir(dir, base = dir) {
  const results = [];
  for (const name of fs.readdirSync(dir)) {
    const full = path.join(dir, name);
    const stat = fs.statSync(full);
    const rel = path.relative(base, full).replace(/\\/g, '/');
    if (stat.isDirectory()) {
      results.push(...walkDir(full, base));
    } else {
      results.push({ path: rel, size: stat.size, mtime: stat.mtime });
    }
  }
  return results;
}

// GET /api/zip-preview?dir=uploads
app.get('/api/zip-preview', (req, res) => {
  const dir = path.join(__dirname, req.query.dir || 'uploads');
  if (!dir.startsWith(__dirname)) return res.status(400).json({ error: 'Invalid path' });

  const files = walkDir(dir);
  const totalSize = files.reduce((s, f) => s + f.size, 0);

  res.json({
    dir: req.query.dir,
    fileCount: files.length,
    totalSize,                        // bytes
    files
  });
});
